1. Récupérer un argument N pour le nombre de threads

#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>

int main(int argc, char *argv[]) {
    if (argc != 2) {
        fprintf(stderr, "Usage: %s <nombre_de_threads>\n", argv[0]);
        return EXIT_FAILURE;
    }

    int N = atoi(argv[1]);  // Convertit l'argument en entier
    printf("Nombre de threads à créer : %d\n", N);

}

for (int i = 1; i < argc; i++) {
    int status;
    pid_t pid = wait(&status);
    if (pid > 0 && WIFEXITED(status) && WEXITSTATUS(status) == 0) {
        printf("convert %s success\n", argv[i]);
        fflush(stdout);
    }
}

6. Stocke une entrée taper par l'user dans un fichier :

#define OUTPUT_FILE "output.txt"
#define MAX_LENGTH 256

int main() {
    int fd = open(OUTPUT_FILE, O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (fd == -1) {
        perror("open");
        return EXIT_FAILURE;
    }

    char buffer[MAX_LENGTH];

    while (1) {
        printf("Saisir : ");
        if (fgets(buffer, MAX_LENGTH, stdin) == NULL) {
            break;  // Sortir si l'entrée est interrompue (Ctrl+D)
        }

        write(fd, buffer, strlen(buffer));
    }

    close(fd);
}
