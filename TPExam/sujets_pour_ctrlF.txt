5)Le fait d'utiliser IPC_CREAT en conjonction avec IPC_EXCL permet de réinitialiser le semaphore/la section de mémoire partagée 
au moment de sa création. Veuillez choisir une réponse. 
1. VRAI 
2. FAUX ✅

FAUX

Explication :
L'utilisation de IPC_CREAT | IPC_EXCL avec shmget (pour la mémoire partagée) ou semget (pour les sémaphores) ne réinitialise pas une ressource IPC existante, mais empêche simplement sa recréation si elle existe déjà.

IPC_CREAT : Crée un nouvel objet IPC (mémoire partagée, sémaphore, etc.) s'il n'existe pas encore.
IPC_EXCL : Utilisé avec IPC_CREAT, il force l'échec si la ressource existe déjà.
Si la ressource existe déjà, shmget() ou semget() retournera une erreur (EEXIST) au lieu de la réinitialiser.

Cas pratique :
Si vous voulez forcer une réinitialisation, vous devez :

Supprimer la ressource existante avec shmctl(shmid, IPC_RMID, NULL); ou semctl(semid, 0, IPC_RMID);
Recréer la ressource proprement.

dijkstra.c :
#include <sys/sem.h>
#include <stdlib.h>
#include <stdio.h>


#include "dijkstra.h"


int sem_create(key_t cle, int initval) {
        int semid;
        union semun { int val; struct semid_ds *buf; ushort *array; } arg_ctl;


        semid = semget(cle, 1 , IPC_CREAT|IPC_EXCL|0666);
        if (semid == -1)
        {
                semid = semget(cle,1, 0666);
                if (semid == -1)
                {
                        perror("Erreur semget()");
                        exit(1);
                }
        }
        else
        {
                arg_ctl.val = initval;
                if (semctl(semid, 0, SETVAL, arg_ctl) == -1)
                {
                        perror("Erreur initialisation sémaphore");
                        exit(1);
                }
        }
        return(semid) ;
}


void P(int semid)
{
        struct sembuf sempar;
        sempar.sem_num = 0;
        sempar.sem_op = -1;
        sempar.sem_flg = 0;
        if (semop(semid, &sempar, 1) == -1)
                perror("Erreur operation P");
}


void V(int semid)
{
        struct sembuf sempar;
        sempar.sem_num = 0;
        sempar.sem_op = 1;
        sempar.sem_flg = 0;
        if (semop(semid, &sempar, 1) == -1)
                perror("Erreur opération V");
}


void sem_delete(int semid)
{
        if (semctl(semid,0,IPC_RMID,0) == -1)
                perror("Erreur dans destruction sémaphore");
}

Question 1 : Quelles sont les affirmations vraies concernant cette instruction: int val = dup2(tube[0],0);
Veuillez choisir au moins une réponse.
	a. dup2 inverse les fichiers pointés par les descripteurs de fichier. tube[0] pointe vers l'entrée standard et 0 pointe sur la sortie du tube
	b. cette instruction contient une erreur
	c. dup2 ferme automatiquement le fichier pointé par 0
	d. dup2 copie la valeur 0 dans la variable tube[0]
	e. dup2 ferme automatiquement le fichier pointé par tube[0]
	f. dup2 renvoie tube[0] en cas de succès
	g. tube[0] et 0 pointent sur le même fichier après cette instruction
	
La fonction dup2(oldfd, newfd) copie le descripteur de fichier oldfd vers newfd, en fermant automatiquement newfd s'il était déjà ouvert.

Analysons chaque affirmation :

a. ❌ Faux. dup2 ne "renverse" pas les fichiers pointés. Il redirige newfd (ici 0, qui correspond à stdin) vers oldfd (tube[0]). Après l'exécution, tube[0] et 0 désigneront le même fichier.

b. ❌ Faux. Il n'y a pas d'erreur syntaxique ou logique dans l'instruction.

c. ✅ Vrai. dup2 ferme automatiquement newfd (0, soit stdin) s'il était ouvert avant d'effectuer la redirection.

d. ❌ Faux. dup2 ne modifie pas la valeur de tube[0], il redirige simplement 0 vers tube[0].

e. ❌ Faux. dup2 ne ferme pas tube[0]. Il ferme uniquement newfd s'il était ouvert avant la duplication.

f. ✅ Vrai. dup2 retourne newfd en cas de succès, donc ici, il renverra 0.

g. ✅ Vrai. Après l'exécution de dup2(tube[0], 0);, tube[0] et 0 (stdin) pointeront vers le même fichier.

Question 2 : L'implémentation de la commande suivante nécessite l'utilisation de 4 tubes.

cut -f 1,3 -d : < /etc/passwd | sed 's+^\(.*\):\(.*\)+\2:\1+' | sort -n > users

Faux 


Analyse des tubes :
Chaque | (pipe) entre les commandes implique l'utilisation d'un tube pour connecter la sortie standard d'une commande à l'entrée standard de la suivante.
Ici, il y a deux |, donc deux tubes.

Question 3 : L'opération P est bloquante lorsque le nombre de jetons dans le semaphore est de 0.
Veuillez choisir une réponse.
Vrai
Faux


Si le sémaphore est strictement positif, l'opération P diminue le compteur et le processus continue.
Si le sémaphore est à 0, l'opération P est bloquante : le processus est mis en attente jusqu'à ce qu'un autre processus effectue une opération V (verhogen, "augmenter"), qui incrémente le sémaphore et libère un jeton.
Donc, l'affirmation est vraie.

Réponse correcte : ✅ Vrai.

Question 4 : Combien de processus (père compris) sont créés par ce programme?

int main() {
    int pid1 , pid2 ;
    pid1 = fork() ;
    pid2 = fork() ;
    if ( (pid1 != 0) && (pid2 == 0) )
    {
        fork() ;
    }
    fork() ;
    return 0;
}

------
if ((pid1 != 0) && (pid2 == 0)) {
    fork();
}

Question 4 : L'ensemble des threads d'un processus possède le même espace mémoire leur permettant d'échanger des données.
Veuillez choisir une réponse.
Vrai
Faux


Dans un programme multi-thread :

Tous les threads d’un même processus partagent le même espace mémoire.
Ils ont accès aux mêmes variables globales, aux mêmes segments de données (heap, stack), et peuvent modifier les mêmes structures.

Pourquoi ?
Les threads d’un même processus partagent :

La mémoire globale (heap, data, text segment).
Les descripteurs de fichiers.
Les mappages mémoire (mmap).

Chaque thread a cependant :

Son propre contexte d’exécution (registre, compteur d'instruction).
Sa propre pile (stack), isolée des autres.

Conséquences
✅ Avantage : Communication rapide entre threads (pas besoin d'IPC, de pipes, etc.).
❌ Inconvénient : Risque de conditions de course (si plusieurs threads modifient la même variable sans synchronisation).
💡 Utilisation des mutex/semaphores est nécessaire pour éviter les conflits lors de l'accès aux données partagées.

👉 Conclusion : Tous les threads d'un processus partagent la mémoire du processus, donc la réponse est Vrai.✅✅ 🚀

Comportement de exit() dans un programme multithread
Si un thread appelle exit(), cela :

Termine immédiatement le processus entier, y compris tous les threads.
Ferme les fichiers ouverts et libère les ressources.
Retourne le code de sortie (status) au système d'exploitation.
💡 En gros, exit() tue tout le programme, pas seulement le thread qui l'appelle.

     {
         P(sem2);
         char c = shmem[i%MEM];
         V(sem1);
         printf("%c",c);
     }

     sem_delete(sem1);
     sem_delete(sem2);

📌 Pourquoi l'exécution n'est pas déterministe ? (Explication simple)
Un programme est déterministe si on obtient toujours exactement le même résultat, peu importe l’exécution.

Ici, le producteur (prod) et le consommateur (conso) tournent en parallèle, et c’est le système d’exploitation qui décide quand chaque processus s’exécute.

Comme prod et conso ne sont pas parfaitement synchronisés sur l'affichage, l’ordre exact des affichages peut varier.

Le printf() est bufferisé → il n'affiche pas immédiatement les caractères.

SEM=5 et MEM=3 donc comme shmem[i%MEM] est un buffer circulaire, il peut écraser les lettres qu'il a écrit avant que conso les lise donc
le résultat peut varier.

int xy_to_index(int x, int y, size_t width)
{
    return (y * 3 * width) + (3 * x);
}


char* readline(int fd)
{
    int offset = 0;
    size_t size;
    char dummy;
    do {
        size = read(fd,&dummy,1);
        offset += size;
    } while(size != 0 && dummy != '\n' && dummy != EOF);


    char* line = malloc(offset * sizeof(char)+1);
    lseek(fd,-offset,SEEK_CUR);
    read(fd,line,offset);
    line[offset]='\0';
    return line;
}


void load_image(char* filepath, unsigned char** image_data, size_t* width, size_t* height)
{
    int fd = open(filepath,O_RDONLY);
    if (fd == -1)
    {
        perror("Can't open input file");
        exit(-1);
    }
    char* line = readline(fd);
    if ( strncmp("P6", line, 2) != 0)
    {
        fputs("Input File is not a ppm file\n", stderr);
        free(line);
        exit(-1);
    }
    free(line);
    do {
        line = readline(fd);
        if (line[0] == '#')
        {
            free(line);
            continue;
        }
        int success = sscanf(line,"%lu %lu",width,height);
        if (success != 2)
        {
            fputs("Input File is not a valid ppm file\n", stderr);
            free(line);
            exit(-1);
        }
        free(line);
    } while(*width == 0);
    int dummy = 0;
    do {
        line = readline(fd);
        if (line[0] == '#')
        {
            free(line);
            continue;
        }
        int success = sscanf(line,"%d",&dummy);
        if (success != 1)
        {
            fputs("Input File is not a valid ppm file\n", stderr);
            free(line);
            exit(-1);
        }
        free(line);
    } while(dummy == 0);


    unsigned data_size = 3 * (*width) * (*height);
    *image_data = malloc(data_size * sizeof(unsigned char));


    read(fd,*image_data,3*(*width)*(*height));
}


void save_image(char* filepath, unsigned char* image, size_t width, size_t height)
{
    int fd = open(filepath,O_CREAT|O_WRONLY,0644);
    if (fd == -1)
    {
        perror("Can't open output file");
        exit(-1);
    }
    write(fd,"P6\n",3);
    char* size = NULL;
    asprintf(&size,"%lu %lu\n",width, height);
    write(fd,size,strlen(size));
    free(size);
    write(fd,"255\n",4);
    write(fd,image,3*width*height);
    write(fd,"\n",1);
    close(fd);


}


void calcul(unsigned char* image,unsigned char* out_image, int width, int height, int x, int y)
{
    int idx = xy_to_index(x,y,width);
    int red = image[idx];
    int green = image[idx+1];
    int blue = image[idx+2];
    int new_red = 0;
    int new_green = 0;
    int new_blue = 0;
    for (int i = 0; i <= 1000; i++)
    {
        float fact = 393.0f / i;
        new_red =   fact*red + 0.769*green + 0.189*blue;
        fact = 686.0f/i;
        new_green = 0.349*red + fact*green + 0.168*blue;
        fact = 131.0f/i;
        new_blue =  0.272*red + 0.534*green + fact*blue;
    }
    out_image[idx] = new_red <=  255 ? new_red : 255;
    out_image[idx+1] = new_green <= 255 ? new_green  : 255;
    out_image[idx+2] = new_blue <= 255 ? new_blue : 255;
}
